  <script>
    // --- State & Config ---
    const socket = io();
    let myStream;
    let myName = '';
    let myRoom = '';
    let isHost = false;
    let isMicOn = false;
    let isCamOn = false;
    const peers = {}; // WebRTC Peer Connections
    const video = document.getElementById('mainVideo');

    // --- UI Logic ---
    function showJoinModal() {
      document.getElementById('joinContainer').classList.toggle('hidden');
    }

    // --- Room Management ---
    function createRoom() {
      myName = document.getElementById('usernameInput').value || 'Host';
      socket.emit('create-room', { name: myName });
    }

    function joinRoom() {
      myName = document.getElementById('usernameInput').value || 'Guest';
      const roomId = document.getElementById('roomCodeInput').value;
      if(!roomId) return alert('Enter a code');
      socket.emit('join-room', { roomId: roomId, name: myName });
    }

    socket.on('room-created', (data) => {
      isHost = true;
      myRoom = data.roomId;
      showWatchRoom(data.roomId);
    });

    socket.on('room-joined', (data) => {
      myRoom = data.roomId;
      showWatchRoom(data.roomId);
      if(data.videoSrc) {
        video.src = data.videoSrc;
      }
    });

    socket.on('error', (msg) => alert(msg));

    function showWatchRoom(roomId) {
      document.getElementById('entryPage').classList.add('hidden');
      document.getElementById('watchRoom').classList.remove('hidden');
      document.getElementById('displayRoomCode').innerText = roomId;
      startMedia(); // Start Camera/Mic
    }

    function leaveRoom() {
      window.location.reload();
    }

    // --- Video Source Management (Host Driven) ---
    function loadLink() {
      if(!isHost) return alert('Only host can change video');
      const src = document.getElementById('videoLinkInput').value;
      video.src = src;
      socket.emit('change-src', { src: src });
    }

    function uploadVideo(input) {
      if(!isHost) return alert('Only host can upload');
      if (input.files && input.files[0]) {
        const formData = new FormData();
        formData.append('video', input.files[0]);

        fetch('/upload', { method: 'POST', body: formData })
          .then(res => res.json())
          .then(data => {
            video.src = data.filePath;
            socket.emit('change-src', { src: data.filePath });
          })
          .catch(err => console.error(err));
      }
    }

    socket.on('update-src', (data) => {
      video.src = data.src;
    });

    // --- Video Sync ---
    let ignoreNextSeek = false;
    video.addEventListener('play', () => {
      socket.emit('sync-video', { type: 'play', time: video.currentTime });
    });
    video.addEventListener('pause', () => {
      socket.emit('sync-video', { type: 'pause', time: video.currentTime });
    });
    video.addEventListener('seeked', () => {
      if(ignoreNextSeek) { ignoreNextSeek = false; return; }
      socket.emit('sync-video', { type: 'seek', time: video.currentTime });
    });

    socket.on('sync-video', (data) => {
      ignoreNextSeek = true;
      if(data.type === 'play') { video.play(); }
      if(data.type === 'pause') { video.pause(); }
      video.currentTime = data.time;
    });

    // --- Chat ---
    function sendMessage() {
      const input = document.getElementById('chatInput');
      if(!input.value.trim()) return;
      socket.emit('chat-message', { name: myName, msg: input.value });
      input.value = '';
    }

    socket.on('chat-message', (data) => {
      const div = document.createElement('div');
      div.className = 'text-sm p-2 rounded bg-white/5';
      div.innerHTML = `<span class="font-bold text-teal-400">${data.name}:</span> ${data.msg}`;
      document.getElementById('chatBox').appendChild(div);
      document.getElementById('chatBox').scrollTop = 10000;
    });

    // --- WebRTC Call System ---
    async function startMedia() {
      try {
        myStream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });
        // We don't show our own video in the main grid for space, but logic is here
      } catch(e) {
        console.log("Media permission denied");
      }
    }

    function toggleMic() {
      if(!myStream) return;
      const track = myStream.getAudioTracks()[0];
      if(track) track.enabled = !track.enabled;
      isMicOn = !isMicOn;
      document.getElementById('micOn').classList.toggle('hidden', !isMicOn);
      document.getElementById('micOff').classList.toggle('hidden', isMicOn);
      document.getElementById('micBtn').classList.toggle('active', isMicOn);
    }

    function toggleCam() {
      if(!myStream) return;
      const track = myStream.getVideoTracks()[0];
      if(track) track.enabled = !track.enabled;
      isCamOn = !isCamOn;
      document.getElementById('camOn').classList.toggle('hidden', !isCamOn);
      document.getElementById('camOff').classList.toggle('hidden', isCamOn);
      document.getElementById('camBtn').classList.toggle('active', isCamOn);
    }
    
    function toggleChatVisibility() {
      const sidebar = document.getElementById('sidebar');
      sidebar.classList.toggle('hidden');
      document.getElementById('chatToggleBtn').classList.toggle('active');
    }

    // --- WebRTC Signaling (Simplified) ---
    // Note: A full WebRTC implementation with Socket.io requires ~100 lines of code for Offer/Answer/ICE Candidate handling.
    // This is a simplified placeholder logic connecting to the server events.
    
    socket.on('user-joined', async (user) => {
      console.log("User joined:", user.id);
      // Logic to create WebRTC Offer would go here
      // For brevity, this code focuses on the UI and Video Sync.
      // You would use 'peers[user.id] = new RTCPeerConnection...' here.
      
      const div = document.createElement('div');
      div.className = 'p-2 text-xs bg-black/50 rounded';
      div.innerText = `${user.name} joined`;
      document.getElementById('chatBox').appendChild(div);
    });

    socket.on('signal', (data) => {
      // Logic to handle WebRTC Signal (Answer/ICE) would go here
    });

  </script>
